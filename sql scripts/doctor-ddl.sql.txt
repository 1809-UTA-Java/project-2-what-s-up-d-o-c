DROP TABLE DOCTORS;
DROP SEQUENCE D_Sequence;

CREATE TABLE DOCTORS (
	DOCTOR_ID NUMBER(*,0) PRIMARY KEY,
	FIRST_NAME VARCHAR2(30),
	MIDDLE_NAME VARCHAR2(30),
	LAST_NAME VARCHAR2(30),
	DATE_OF_BIRTH DATE,
	SPECIALTY_NAME VARCHAR2(30)
);

CREATE SEQUENCE D_Sequence START WITH 11 INCREMENT BY 1 NOCACHE;
