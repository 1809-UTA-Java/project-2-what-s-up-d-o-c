CREATE TABLE DRUGS (
	id integer primary key,
	name varchar2(25),
	description varchar2(100),
	inventory integer check (d_inventory >= 0)
);

CREATE SEQUENCE drugs_seq
 START WITH 1
 INCREMENT BY 1
 NOCACHE;
