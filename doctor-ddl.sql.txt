DROP TABLE DOCTORS;
DROP TABLE NAMES;
DROP TABLE SPECIALTIES;
DROP SEQUENCE D_Sequence;
DROP SEQUENCE N_Sequence;
DROP SEQUENCE S_Sequence;

CREATE TABLE DOCTORS (
	DOCTOR_ID NUMBER(*,0) PRIMARY KEY,
	NAME NUMBER(*,0),
	DATE_OF_BIRTH DATE,
	SPECIALTY NUMBER(*,0)
);

CREATE TABLE NAMES (
	NAME_ID NUMBER(*,0) PRIMARY KEY,
	FIRST_NAME VARCHAR2(30),
	MIDDLE_NAME VARCHAR2(30),
	LAST_NAME VARCHAR2(30)
);

CREATE TABLE SPECIALTIES(
	SPECIALTY_ID NUMBER(*,0)PRIMARY KEY,
	SPECIALTY_NAME VARCHAR2(30)
);

CREATE SEQUENCE D_Sequence START WITH 1 INCREMENT BY 1;
CREATE SEQUENCE N_Sequence START WITH 1 INCREMENT BY 1;
CREATE SEQUENCE S_Sequence START WITH 1 INCREMENT BY 1;

ALTER TABLE DOCTORS ADD CONSTRAINT DOCTOR_NAME_FK
FOREIGN KEY (NAME) REFERENCES NAMES (NAME_ID);

ALTER TABLE DOCTORS ADD CONSTRAINT DOCTOR_SPECIALTY_FK
FOREIGN KEY (SPECIALTY) REFERENCES SPECIALTIES (SPECIALTY_ID);
